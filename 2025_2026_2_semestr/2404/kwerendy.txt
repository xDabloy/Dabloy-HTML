1. SELECT marka, model, cena FROM pojazdy WHERE marka = "BM" ORDER BY cena DESC LIMIT 15 ;
2. SELECT
    (SELECT AVG(cena) AS "Srednia_cena" FROM pojazdy),
   	(SELECT MAX(cena) FROM pojazdy WHERE model = 'meta') As "Maksymalna_cena";

3.SELECT
    p.marka,
    p.model,
    p.cena,
    k.nazwa,
    k.doplata
FROM
    pojazdy p
JOIN kolory k ON
    p.kolor = k.id
WHERE
    p.model = 'alfa';
4.SELECT marka, model, cena FROM pojazdy ORDER BY RAND() LIMIT 2;
